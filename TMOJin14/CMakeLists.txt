CMAKE_MINIMUM_REQUIRED(VERSION 2.4.8)
SET(CMAKE_COLOR_MAKEFILE ON)
SET(CMAKE_VERBOSE_MAKEFILE ON)
SET(CMAKE_INCLUDE_CURRENT_DIR TRUE)

IF(UNIX AND NOT LINUX_SET)
 ADD_DEFINITIONS(-D LINUX)
ENDIF(UNIX AND NOT LINUX_SET)

FIND_PACKAGE(CGAL REQUIRED)
IF (CGAL_FOUND)
  MESSAGE("CGAL found OK")
ELSE(CGAL_FOUND)
  MESSAGE(FATAL_ERROR "CGAL library not found.")
ENDIF(CGAL_FOUND)

INCLUDE_DIRECTORIES(
${CGAL_INCLUDE_DIRS}
../tmolib/
.
)

LINK_LIBRARIES (
${CGAL_LIBRARY}
${CGAL_LIBRARIES} 
${CGAL_3RD_PARTY_LIBRARIES} 
tmo
)

SET(TMO_SOURCES
 TMOJin14.cpp
 TMOPlugin.cpp
)


SET(TMOWARD_HEADERS
 TMOJin14.h
 TMOPlugin.h
)


add_library( jin14 SHARED ${TMO_SOURCES} )
add_custom_command( TARGET jin14 POST_BUILD 
	COMMAND cp -f libjin14.so ../TMOCmd/jin14.tml
	COMMAND cp -f libjin14.so ../TMOgui/jin14.tml 
	COMMAND cp -f libjin14.so ../jin14.tml 
#	COMMAND rm -f libjin14.so 
	)
